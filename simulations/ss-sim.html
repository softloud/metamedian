get_effect <- function(this_tibble) {
  metamedian::effect_sd(
    centre = this_tibble$median,
    spread = this_tibble$iqr,
    n = this_tibble$n,
    centre_type = "median",
    spread_type = "iqr"
  )
}

ss_sim <- ss_data %>%
